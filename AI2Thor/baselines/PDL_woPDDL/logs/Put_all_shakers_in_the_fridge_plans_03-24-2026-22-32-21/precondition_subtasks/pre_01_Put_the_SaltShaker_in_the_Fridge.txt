# GENERAL TASK DECOMPOSITION 
# Decompose and parallel subtasks where ever possible
# Independent subtasks:

# Initial condition analyze due to previous subtask:
# 1. Robot not holding SaltShaker.
# 2. Robot not at SaltShaker location.
# 3. Fridge is initially closed.

# SubTask 1: Put the SaltShaker in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: saltShaker(Location=counterTop), fridge(Location=floor)

GoToObject(robot1, saltShaker)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 saltShaker)

PickupObject(robot1, saltShaker, counterTop)
   Preconditions:
     (at-location saltShaker counterTop)
     (at robot1 saltShaker)
     (not (inaction robot1))
   Effects:
     (holding robot1 saltShaker)

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (is-fridge fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, saltShaker, fridge)
   Preconditions:
     (holding robot1 saltShaker)
     (at robot1 fridge)
     (not (inaction robot1))
     (is-fridge fridge)
     (fridge-open fridge)
   Effects:
     (at-location saltShaker fridge)
     (not (holding robot1 saltShaker))
     (not (inaction robot1))

CloseFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
     (is-fridge fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location saltShaker fridge)
(not (fridge-open fridge))
