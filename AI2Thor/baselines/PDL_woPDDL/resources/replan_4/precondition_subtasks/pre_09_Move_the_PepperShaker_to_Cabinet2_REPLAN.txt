# Initial condition analyze due to previous subtask:
#1. Robot1 is holding Apple
#2. PepperShaker is not found in the scene
#3. Cabinet2 is not full and located on the floor

# SubTask 4: Move the PepperShaker to Cabinet2.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: pepperShaker(Location=floor), cabinet2(Location=floor)

GoToObject(robot1, pepperShaker)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 pepperShaker)

PickupObject(robot1, pepperShaker, floor)
   Preconditions:
     (at-location pepperShaker floor)
     (at robot1 pepperShaker)
     (not (inaction robot1))
   Effects:
     (holding robot1 pepperShaker)

GoToObject(robot1, cabinet2)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 cabinet2)

OpenObject(robot1, cabinet2)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet2)
   Effects:
     (object-open robot1 cabinet2)

PutObject(robot1, pepperShaker, cabinet2)
   Preconditions:
     (holding robot1 pepperShaker)
     (at robot1 cabinet2)
     (not (inaction robot1))
     (not (object-close robot1 cabinet2))
   Effects:
     (at-location pepperShaker cabinet2)
     (not (holding robot1 pepperShaker))

CloseObject(robot1, cabinet2)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet2)
     (object-open robot1 cabinet2)
   Effects:
     (object-close robot1 cabinet2)

Goal condition:
(at-location pepperShaker cabinet2)
(object-close robot1 cabinet2)
