# SubTask 10: Place the Bread in Cabinet3.
   Skills Required: GoToObject, OpenObject, PutObject, CloseObject
   Related Objects: bread(Location=robot1), cabinet3(Location=floor)

GoToObject(robot1, cabinet3)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 cabinet3)

OpenObject(robot1, cabinet3)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet3)
   Effects:
     (object-open robot1 cabinet3)
     (not (object-close robot1 cabinet3))

PutObject(robot1, bread, cabinet3)
   Preconditions:
     (holding robot1 bread)
     (at robot1 cabinet3)
     (not (inaction robot1))
     (not (object-close robot1 cabinet3))
   Effects:
     (at-location bread cabinet3)
     (not (holding robot1 bread))
     (not (inaction robot1))

CloseObject(robot1, cabinet3)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet3)
     (object-open robot1 cabinet3)
   Effects:
     (object-close robot1 cabinet3)
     (not (object-open robot1 cabinet3))

Goal condition:
(at-location bread cabinet3)
