# SubTask 6: Move the Mug to the Appropriate Location.
    Skills Required: GoToObject, PickupObject, PutObject
    Related Objects: mug(Location=unknown), cabinet1(Location=floor)

GoToObject(robot1, mug)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 mug)

PickupObject(robot1, mug, unknown)
   Preconditions:
     (at-location mug unknown)
     (at robot1 mug)
     (not (inaction robot1))
   Effects:
     (holding robot1 mug)
     (not (at-location mug unknown))

GoToObject(robot1, cabinet1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 cabinet1)

PutObject(robot1, mug, cabinet1)
   Preconditions:
     (holding robot1 mug)
     (at robot1 cabinet1)
     (not (inaction robot1))
     (not (is-fridge cabinet1))
     (not (object-close robot1 cabinet1))
   Effects:
     (at-location mug cabinet1)
     (not (holding robot1 mug))
     (not (inaction robot1))

Goal condition:
(at-location mug cabinet1)
