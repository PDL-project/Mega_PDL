# SubTask 1: Place the SaltShaker in Cabinet1.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: saltShaker(Location=unknown), cabinet1(Location=floor)

GoToObject(robot1, saltShaker)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 saltShaker)

PickupObject(robot1, saltShaker, countertop1)
   Preconditions:
     (at-location saltShaker countertop1)
     (at robot1 saltShaker)
     (not (inaction robot1))
     (not (object-close robot1 countertop1))
   Effects:
     (holding robot1 saltShaker)
     (not (at-location saltShaker countertop1))

GoToObject(robot1, cabinet1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 cabinet1)

OpenObject(robot1, cabinet1)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet1)
   Effects:
     (object-open robot1 cabinet1)
     (not (object-close robot1 cabinet1))

PutObject(robot1, saltShaker, cabinet1)
   Preconditions:
     (holding robot1 saltShaker)
     (at robot1 cabinet1)
     (not (inaction robot1))
     (not (object-close robot1 cabinet1))
   Effects:
     (at-location saltShaker cabinet1)
     (not (holding robot1 saltShaker))

CloseObject(robot1, cabinet1)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet1)
     (object-open robot1 cabinet1)
   Effects:
     (object-close robot1 cabinet1)
     (not (object-open robot1 cabinet1))

Goal condition: 
(at-location saltShaker cabinet1)
(object-close robot1 cabinet1)
