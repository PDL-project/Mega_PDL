# SubTask 3: Place the Fork in Drawer3.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: fork(Location=countertop2), drawer3(Location=floor)

GoToObject(robot1, countertop2)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 countertop2)

PickupObject(robot1, fork, countertop2)
   Preconditions:
     (at-location fork countertop2)
     (at robot1 fork)
     (not (inaction robot1))
   Effects:
     (holding robot1 fork)
     (not (at-location fork countertop2))

GoToObject(robot1, drawer3)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 drawer3)

OpenObject(robot1, drawer3)
   Preconditions:
     (not (inaction robot1))
     (at robot1 drawer3)
   Effects:
     (object-open robot1 drawer3)
     (not (object-close robot1 drawer3))

PutObject(robot1, fork, drawer3)
   Preconditions:
     (holding robot1 fork)
     (at robot1 drawer3)
     (not (inaction robot1))
     (not (object-close robot1 drawer3))
   Effects:
     (at-location fork drawer3)
     (not (holding robot1 fork))

CloseObject(robot1, drawer3)
   Preconditions:
     (not (inaction robot1))
     (at robot1 drawer3)
     (object-open robot1 drawer3)
   Effects:
     (object-close robot1 drawer3)
     (not (object-open robot1 drawer3))

Goal condition: 
(at-location fork drawer3)
