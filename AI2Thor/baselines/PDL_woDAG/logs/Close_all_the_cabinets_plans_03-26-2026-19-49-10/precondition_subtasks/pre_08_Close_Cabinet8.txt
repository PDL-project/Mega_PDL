# SubTask 8: Close Cabinet8.
    Skills Required: GoToObject, CloseObject
    Related Objects: cabinet8(Location=floor)

# Initial condition:
#1. Robot not at Cabinet8 location
#2. Cabinet8 is open

GoToObject(robot1, cabinet8)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 cabinet8)

CloseObject(robot1, cabinet8)
   Preconditions:
     (not (inaction robot1))
     (at robot1 cabinet8)
   Effects:
     (object-close robot1 cabinet8)
     (not (object-open robot1 cabinet8))

Goal condition:
(object-close robot1 cabinet8)
