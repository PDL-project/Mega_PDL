# SubTask 2: Store the Pen on the SideTable.
    Skills Required: GoToObject, PickupObject, PutObject
    Related Objects: pen(Location=sofa), sidetable2(Location=floor)

# Initial condition:
#1. Robot not holding Pen
#2. Sofa is the initial location of Pen

GoToObject(robot1, pen)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 pen)

PickupObject(robot1, pen, sofa)
   Preconditions:
     (at-location pen sofa)
     (at robot1 pen)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 pen)
     (not (hand-empty robot1))
     (not (at-location pen sofa))

GoToObject(robot1, sidetable2)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 sidetable2)

PutObject(robot1, pen, sidetable2)
   Preconditions:
     (holding robot1 pen)
     (at robot1 sidetable2)
     (not (inaction robot1))
   Effects:
     (at-location pen sidetable2)
     (not (holding robot1 pen))
     (hand-empty robot1)

Goal condition:
(at-location pen sidetable2)
