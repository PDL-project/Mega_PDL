# SubTask 11: Move the Tomato to the appropriate location.
    Skills Required: GoToObject, PickupObject, PutObject
    Related Objects: tomato(Location=CounterTop), fridge(Location=Floor)

   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 tomato)

PickupObject(robot1, tomato, CounterTop)
   Preconditions:
     (at-location tomato CounterTop)
     (at robot1 tomato)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 tomato)
     (not (hand-empty robot1))
     (not (at-location tomato CounterTop))

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

PutObject(robot1, tomato, fridge)
   Preconditions:
     (holding robot1 tomato)
     (at robot1 fridge)
     (not (inaction robot1))
     (not (is-fridge fridge))
     (not (object-close robot1 fridge))
   Effects:
     (at-location tomato fridge)
     (not (holding robot1 tomato))
     (hand-empty robot1)
