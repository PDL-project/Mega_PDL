# Independent subtasks:

# Initial condition analyze due to previous subtask:
#1. Robot not holding Fork
#2. Drawer1 is closed

# SubTask 1: Put the Fork in Drawer1.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: Fork(Location=CounterTop), Drawer1(Location=Floor)

GoToObject(robot1, Fork)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Fork)

PickupObject(robot1, Fork, CounterTop)
   Preconditions:
     (at-location Fork CounterTop)
     (at robot1 Fork)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 Fork)
     (not (hand-empty robot1))
     (not (at-location Fork CounterTop))

GoToObject(robot1, Drawer1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Drawer1)

OpenObject(robot1, Drawer1)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Drawer1)
   Effects:
     (object-open robot1 Drawer1)
     (not (object-close robot1 Drawer1))

PutObject(robot1, Fork, Drawer1)
   Preconditions:
     (holding robot1 Fork)
     (at robot1 Drawer1)
     (not (inaction robot1))
     (not (object-close robot1 Drawer1))
   Effects:
     (at-location Fork Drawer1)
     (not (holding robot1 Fork))
     (hand-empty robot1)

CloseObject(robot1, Drawer1)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Drawer1)
     (object-open robot1 Drawer1)
   Effects:
     (object-close robot1 Drawer1)
     (not (object-open robot1 Drawer1))

Goal condition: 
(at-location Fork Drawer1)
(object-close robot1 Drawer1)
