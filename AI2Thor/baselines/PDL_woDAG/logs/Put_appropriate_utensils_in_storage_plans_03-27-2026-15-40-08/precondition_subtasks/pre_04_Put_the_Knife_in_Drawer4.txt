# SubTask 4: Put the Knife in Drawer4.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: Knife(Location=CounterTop), Drawer4(Location=Floor)

GoToObject(robot1, Knife)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Knife)

PickupObject(robot1, Knife, CounterTop)
   Preconditions:
     (at-location Knife CounterTop)
     (at robot1 Knife)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 Knife)
     (not (hand-empty robot1))
     (not (at-location Knife CounterTop))

GoToObject(robot1, Drawer4)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Drawer4)

OpenObject(robot1, Drawer4)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Drawer4)
   Effects:
     (object-open robot1 Drawer4)
     (not (object-close robot1 Drawer4))

PutObject(robot1, Knife, Drawer4)
   Preconditions:
     (holding robot1 Knife)
     (at robot1 Drawer4)
     (not (inaction robot1))
     (not (object-close robot1 Drawer4))
   Effects:
     (at-location Knife Drawer4)
     (not (holding robot1 Knife))
     (hand-empty robot1)

CloseObject(robot1, Drawer4)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Drawer4)
     (object-open robot1 Drawer4)
   Effects:
     (object-close robot1 Drawer4)
     (not (object-open robot1 Drawer4))

Goal condition:
(at-location Knife Drawer4)
(object-close robot1 Drawer4)
