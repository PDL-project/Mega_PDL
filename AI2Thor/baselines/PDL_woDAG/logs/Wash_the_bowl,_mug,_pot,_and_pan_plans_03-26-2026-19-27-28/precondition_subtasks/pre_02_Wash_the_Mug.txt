# SubTask 2: Wash the Mug.
    Skills Required: GoToObject, PickupObject, CleanObject, PutObject
    Related Objects: mug(Location=counterTop), sink(Location=floor)

GoToObject(robot1, mug)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 mug)

PickupObject(robot1, mug, counterTop)
   Preconditions:
     (at-location mug counterTop)
     (at robot1 mug)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 mug)
     (not (hand-empty robot1))
     (not (at-location mug counterTop))

GoToObject(robot1, sink)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 sink)

SwitchOn(robot1, Faucet)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Faucet)
   Effects:
     (faucet-on)

CleanObject(robot1, mug, sink)
   Preconditions:
     (not (inaction robot1))
     (holding robot1 mug)
     (at robot1 sink)
     (is-sink sink)
     (faucet-on)
   Effects:
     (cleaned robot1 mug)

SwitchOff(robot1, Faucet)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Faucet)
   Effects:
     (not (faucet-on))

GoToObject(robot1, counterTop)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 counterTop)

PutObject(robot1, mug, counterTop)
   Preconditions:
     (holding robot1 mug)
     (at robot1 counterTop)
     (not (inaction robot1))
   Effects:
     (at-location mug counterTop)
     (not (holding robot1 mug))
     (hand-empty robot1)

Goal condition:
