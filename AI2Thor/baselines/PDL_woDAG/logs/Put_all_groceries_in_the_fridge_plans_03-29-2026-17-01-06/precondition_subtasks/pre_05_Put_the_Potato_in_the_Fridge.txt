# SubTask 5: Put the Potato in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObject, CloseObject
    Related Objects: potato(Location=counterTop), fridge(Location=floor)

GoToObject(robot1, potato)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 potato)

PickupObject(robot1, potato, counterTop)
   Preconditions:
     (at-location potato counterTop)
     (at robot1 potato)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 potato)
     (not (hand-empty robot1))
     (not (at-location potato counterTop))

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, potato, fridge)
   Preconditions:
     (holding robot1 potato)
     (at robot1 fridge)
     (not (inaction robot1))
     (fridge-open fridge)
   Effects:
     (at-location potato fridge)
     (not (holding robot1 potato))
     (hand-empty robot1)

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition:
(at-location potato fridge)
(not (fridge-open fridge))
