# SubTask 5: Put the Egg in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObjectInFridge, CloseObject
    Related Objects: egg(Location=fridge), fridge(Location=floor)

GoToObject(robot1, egg)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 egg)

PickupObject(robot1, egg, fridge)
   Preconditions:
     (at-location egg fridge)
     (at robot1 egg)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 egg)
     (not (hand-empty robot1))
     (not (at-location egg fridge))

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, egg, fridge)
   Preconditions:
     (holding robot1 egg)
     (at robot1 fridge)
     (not (inaction robot1))
     (fridge-open fridge)
   Effects:
     (at-location egg fridge)
     (not (holding robot1 egg))
     (hand-empty robot1)

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (fridge-open fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location egg fridge)
(not (fridge-open fridge))
