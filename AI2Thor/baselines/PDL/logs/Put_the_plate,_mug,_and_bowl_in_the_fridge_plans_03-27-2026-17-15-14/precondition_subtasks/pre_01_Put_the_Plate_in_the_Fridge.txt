# SubTask 1: Put the Plate in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObjectInFridge, CloseObject
    Related Objects: plate(Location=CounterTop), fridge(Location=Floor)

GoToObject(robot1, plate)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 plate)

PickupObject(robot1, plate, CounterTop)
   Preconditions:
     (at-location plate CounterTop)
     (at robot1 plate)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 plate)
     (not (hand-empty robot1))
     (not (at-location plate CounterTop))

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, plate, fridge)
   Preconditions:
     (holding robot1 plate)
     (at robot1 fridge)
     (not (inaction robot1))
     (fridge-open fridge)
   Effects:
     (at-location plate fridge)
     (not (holding robot1 plate))
     (hand-empty robot1)

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location plate fridge)
(not (fridge-open fridge))
