# SubTask 3: Put the Bowl in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObjectInFridge, CloseObject
    Related Objects: bowl(Location=CounterTop), fridge(Location=Floor)

GoToObject(robot1, bowl)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 bowl)

PickupObject(robot1, bowl, CounterTop)
   Preconditions:
     (at-location bowl CounterTop)
     (at robot1 bowl)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 bowl)
     (not (hand-empty robot1))
     (not (at-location bowl CounterTop))

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, bowl, fridge)
   Preconditions:
     (holding robot1 bowl)
     (at robot1 fridge)
     (not (inaction robot1))
     (is-fridge fridge)
     (fridge-open fridge)
   Effects:
     (at-location bowl fridge)
     (not (holding robot1 bowl))
     (hand-empty robot1)

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition:
(at-location bowl fridge)
(not (fridge-open fridge))
