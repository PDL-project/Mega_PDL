# SubTask 2: Put the PepperShaker in the Fridge.
    Skills Required: GoToObject, PickupObject, OpenObject, PutObjectInFridge, CloseObject
    Related Objects: pepperShaker(Location=counterTop), fridge(Location=floor)

GoToObject(robot1, pepperShaker)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 pepperShaker)

PickupObject(robot1, pepperShaker, counterTop)
   Preconditions:
     (at-location pepperShaker counterTop)
     (at robot1 pepperShaker)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 pepperShaker)
     (not (hand-empty robot1))
     (not (at-location pepperShaker counterTop))

GoToObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, pepperShaker, fridge)
   Preconditions:
     (holding robot1 pepperShaker)
     (at robot1 fridge)
     (not (inaction robot1))
     (fridge-open fridge)
   Effects:
     (at-location pepperShaker fridge)
     (not (holding robot1 pepperShaker))
     (hand-empty robot1)

CloseObject(robot1, fridge)
   Preconditions:
     (not (inaction robot1))
     (at robot1 fridge)
     (object-open robot1 fridge)
   Effects:
     (object-close robot1 fridge)
     (not (fridge-open fridge))

Goal condition: 
(at-location pepperShaker fridge)
(not (fridge-open fridge))
