# SubTask 4: Put the Knife in Drawer2.

    Related Objects: Knife(Location=CounterTop), Drawer2(Location=Floor)

     (not (inaction robot1))
   Effects:

PickupObject(robot1, Knife, CounterTop)
   Preconditions:
     (at-location Knife CounterTop)
     (at robot1 Knife)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 Knife)
     (not (hand-empty robot1))
     (not (at-location Knife CounterTop))

GoToObject(robot1, Drawer2)
   Preconditions:

   Effects:

OpenObject(robot1, Drawer2)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Drawer2)
   Effects:
     (object-open robot1 Drawer2)
     (not (object-close robot1 Drawer2))

PutObject(robot1, Knife, Drawer2)
   Preconditions:
     (holding robot1 Knife)

     (not (inaction robot1))
     (not (object-close robot1 Drawer2))

     (at-location Knife Drawer2)
     (not (holding robot1 Knife))
     (hand-empty robot1)

CloseObject(robot1, Drawer2)
   Preconditions:
     (not (inaction robot1))
     (at robot1 Drawer2)
