# SubTask 3: Put the RemoteControl on the Sofa.
    Skills Required: GoToObject, PickupObject, PutObject
    Related Objects: RemoteControl(Location=SideTable), Sofa(Location=Floor)

GoToObject(robot1, RemoteControl)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 RemoteControl)

PickupObject(robot1, RemoteControl, SideTable)
   Preconditions:
     (at-location RemoteControl SideTable)
     (at robot1 RemoteControl)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 RemoteControl)
     (not (hand-empty robot1))
     (not (at-location RemoteControl SideTable))

GoToObject(robot1, Sofa)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Sofa)

PutObject(robot1, RemoteControl, Sofa)
   Preconditions:
     (holding robot1 RemoteControl)
     (at robot1 Sofa)
     (not (inaction robot1))
   Effects:
     (at-location RemoteControl Sofa)
     (not (holding robot1 RemoteControl))
     (hand-empty robot1)

Goal condition:
(at-location RemoteControl Sofa)
