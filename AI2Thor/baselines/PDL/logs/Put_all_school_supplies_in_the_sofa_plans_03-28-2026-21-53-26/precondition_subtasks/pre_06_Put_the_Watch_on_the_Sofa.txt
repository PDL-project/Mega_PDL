# SubTask 6: Put the Watch on the Sofa.
    Skills Required: GoToObject, PickupObject, PutObject
    Related Objects: Watch(Location=SideTable), Sofa(Location=Floor)

GoToObject(robot1, Watch)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Watch)

PickupObject(robot1, Watch, SideTable)
   Preconditions:
     (at-location Watch SideTable)
     (at robot1 Watch)
     (not (inaction robot1))
     (hand-empty robot1)
   Effects:
     (holding robot1 Watch)
     (not (hand-empty robot1))
     (not (at-location Watch SideTable))

GoToObject(robot1, Sofa)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Sofa)

PutObject(robot1, Watch, Sofa)
   Preconditions:
     (holding robot1 Watch)
     (at robot1 Sofa)
     (not (inaction robot1))
   Effects:
     (at-location Watch Sofa)
     (not (holding robot1 Watch))
     (hand-empty robot1)

Goal condition:
(at-location Watch Sofa)
