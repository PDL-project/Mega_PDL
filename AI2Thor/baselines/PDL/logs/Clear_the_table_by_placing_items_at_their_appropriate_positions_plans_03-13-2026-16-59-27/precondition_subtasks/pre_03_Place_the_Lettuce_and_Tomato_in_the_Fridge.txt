   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 fridge)

OpenFridge(robot1, fridge)
   Preconditions:
     (not (inaction robot1))

     (is-fridge fridge)
   Effects:
     (object-open robot1 fridge)
     (fridge-open fridge)

PutObjectInFridge(robot1, lettuce, fridge)

   Effects:
     (at-location lettuce fridge)
     (not (holding robot1 lettuce))
     (not (inaction robot1))
