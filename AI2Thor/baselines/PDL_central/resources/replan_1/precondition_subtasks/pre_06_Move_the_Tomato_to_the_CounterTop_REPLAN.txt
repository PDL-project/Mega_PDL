# SubTask 6: Move the Tomato to the CounterTop.
    Skills Required: GoToObject, PickupObject, PutObject
    Related Objects: tomato(Location=kitchen), countertop(Location=kitchen)

GoToObject(robot1, drawer1)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 drawer1)

PutObject(robot1, Bowl, drawer1)
   Preconditions:
     (holding robot1 Bowl)
     (at robot1 drawer1)
     (not (inaction robot1))
     (not (is-fridge drawer1))
     (not (object-close robot1 drawer1))
   Effects:
     (at-location Bowl drawer1)
     (not (holding robot1 Bowl))
     (not (inaction robot1))

GoToObject(robot1, tomato)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 tomato)

PickupObject(robot1, tomato, Floor)
   Preconditions:
     (at-location tomato Floor)
     (at robot1 tomato)
     (not (inaction robot1))
   Effects:
     (holding robot1 tomato)
     (not (at-location tomato Floor))
     (not (inaction robot1))
