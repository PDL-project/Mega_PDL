# SubTask 3: Move the Pencil to the SideTable.
   Skills Required: GoToObject, PickupObject, PutObject
   Related Objects: Pencil(Location=Sofa), SideTable(Location=Floor)

GoToObject(robot1, Pencil)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 Pencil)

PickupObject(robot1, Pencil, Sofa)
   Preconditions:
     (at-location Pencil Sofa)
     (at robot1 Pencil)
     (not (inaction robot1))
   Effects:
     (holding robot1 Pencil)

GoToObject(robot1, SideTable)
   Preconditions:
     (not (inaction robot1))
   Effects:
     (at robot1 SideTable)

PutObject(robot1, Pencil, SideTable)
   Preconditions:
     (holding robot1 Pencil)
     (at robot1 SideTable)
     (not (inaction robot1))
     (not (object-close robot1 SideTable))
   Effects:
     (at-location Pencil SideTable)
     (not (holding robot1 Pencil))
     (not (inaction robot1))

Goal condition:
(at-location Pencil SideTable)
